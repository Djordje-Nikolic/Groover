CREATE KEYSPACE groupChat WITH replication = { 'class' : 'SimpleStrategy', 'replication_factor' : 1};

USE groupChat;

CREATE TABLE messages (
	messageId timeuuid,
	senderId int,
	groupId int,
	createdAt timestamp,
	type text,
	content text,
	image blob,
	trackId timeuuid,
	trackName text,
	trackDuration smallint,
	trackExtension text,
	PRIMARY KEY (groupId, messageId)
) WITH CLUSTERING ORDER BY (messageId DESC);

CREATE TABLE tracksMetadata (
	trackId timeuuid,
	groupId int,
	name text,
	duration smallint,
	format text,
	extension text,
	mimeType text,
	bitrate int,
	hash text,
	chunkCount int,
	PRIMARY KEY (groupId, trackId)
) WITH CLUSTERING ORDER BY (trackId DESC);

CREATE TABLE tracksChunks (
	trackId timeuuid,
	chunkOrder int,
	chunk blob,
	PRIMARY KEY (trackId, chunkOrder)
) WITH CLUSTERING ORDER BY (chunkOrder ASC);
